DROP TABLE IF EXISTS tecnicaDetalhe;
DROP TABLE IF EXISTS servico;
DROP TABLE IF EXISTS cvitem;

DROP TABLE IF EXISTS formacao;
DROP TABLE IF EXISTS servico_item;
DROP TABLE IF EXISTS img;
DROP TABLE IF EXISTS video;
DROP TABLE IF EXISTS contact;
DROP TABLE IF EXISTS destaque;
DROP TABLE IF EXISTS tecnica;
DROP TABLE IF EXISTS funcionario;


CREATE TABLE formacao (
    id SERIAL PRIMARY KEY,
    titulo varchar(50),
    resumo varchar(250),
    texto text,
    data_in date
);

CREATE TABLE destaque (
    id SERIAL PRIMARY KEY,
    titulo varchar(100),
    resumo varchar(500),
    texto text,
	top_flag boolean,
    creation_dt date,
	destaque_dt date
);

CREATE TABLE tecnica (
    id SERIAL PRIMARY KEY,
    nome varchar(100)
);


CREATE TABLE tecnicaDetalhe (
    id SERIAL PRIMARY KEY,
    item varchar(250),
	tecnica_id smallint
);

CREATE TABLE servico (
    id SERIAL PRIMARY KEY,
    pagina varchar(100),
    titulo varchar(100),
    texto varchar
);



CREATE TABLE funcionario (
    id SERIAL PRIMARY KEY,
    nome varchar(50),
    equipa char(2),
    resumo text
);

CREATE TABLE cvitem (
    id SERIAL PRIMARY KEY,
    funcionario_ID smallint ,
    seq int,
	field varchar(40),
	content varchar(500),
	FOREIGN KEY (funcionario_ID) REFERENCES funcionario(id)
);

CREATE TABLE img (
    id SERIAL PRIMARY KEY,
    path varchar(100),
    nome varchar(50),
    descricao varchar(100),
    entidade varchar(30),
    entidade_id smallint
);

CREATE TABLE video (
    id SERIAL PRIMARY KEY,
    url varchar,
    type varchar(10),
    nome varchar(100),
    descricao varchar(200),
    entidade varchar(30),
    entidade_id smallint
);

CREATE TABLE contact (
    id SERIAL PRIMARY KEY,
    email varchar(50),
    nome varchar(50),
    apelido varchar(50),
    telefone varchar(15),
    descricao varchar(1000),
	motivo varchar(10)
);

CREATE TABLE servico_item (
    id SERIAL PRIMARY KEY,
    seq smallint,
	texto varchar(200),
	servico_id int

);
